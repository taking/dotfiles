# ================================
# 📂 파일 / 탐색
# ================================

space f   → 프로젝트 전체 파일 찾기         # 트리 대신 이거로 이동 (기본)
space F   → 현재 폴더 파일 찾기             # 같은 디렉토리 이동 (🔥 매우 자주)
space /   → 전체 검색                       # 문자열/함수 찾기 (🔥 핵심)
space .   → gitignore 토글                  # 숨김 파일 볼 때만 켜기


# ================================
# 🧭 코드 추적 (LSP)
# ================================

g d       → 정의로 이동                     # 함수/변수 정의 위치로 점프
g r       → 참조 찾기                       # 어디서 쓰였는지 확인 (🔥 실전 핵심)


# ================================
# ✏️ 편집 (Helix 핵심 구조)
# ================================

x         → 현재 줄 선택                     # 시작은 항상 selection
c         → 변경 (replace)                  # 선택 영역 수정
d         → 삭제                            # 선택 영역 제거
y         → 복사 (clipboard)               # 시스템 클립보드로 복사

# 🔥 가장 많이 쓰는 조합
x → c     → 한 줄 통째 수정                # 거의 매일 씀
x → d     → 한 줄 삭제
x → y     → 한 줄 복사


# ================================
# 📋 붙여넣기
# ================================

p         → 뒤에 붙여넣기                   # 일반 paste
P         → 선택 덮어쓰기                   # replace paste


# ================================
# 🔁 선택 제어 (Helix 강점)
# ================================

;         → 선택 방향 뒤집기                # multi-selection에서 중요
,         → 마지막 이동 반복                # 반복 작업 속도 ↑
z         → 다음 단어 시작 이동            # 빠른 word 이동


# ================================
# 📏 이동
# ================================

C-u       → 반 페이지 위
C-d       → 반 페이지 아래


# ================================
# 🧼 포맷 / 정리
# ================================

space =   → 코드 포맷                      # 저장 후가 아니라 "필요할 때"


# ================================
# 🧠 상태 정리
# ================================

esc       → 선택 해제                      # 저장 안 함 (auto-save라 필요 없음)


# ================================
# 🚀 실전 흐름 (이게 진짜 중요)
# ================================

# 파일 이동 → 수정
space f → 파일 선택 → x → c

# 사용처 찾기
g r → 리스트 → 이동

# 프로젝트 탐색
space / → 검색 → 이동

# 같은 폴더 이동
space F → 파일 선택
